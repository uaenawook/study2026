# 算数操作符

#### / ：除号

    int main()
    {
        1.对于除号两边都是整数，执行整数除法
    	int ret = 9 / 2;
    	printf("%d\n", ret); //4
    
    	double ret = 9 / 2;
    	printf("%lf\n", ret);//4.0
    
    	2.操作数中有浮点数，才执行浮点数除法
    	double ret = 9 / 2.0;
    	printf("%lf\n", ret);//4.5
    	return 0;
    }

# 位移操作符

    源码 >> 符号不变，其他按位取反 >> 反码
    反码 >> +1 >> 补码

#### << :左移操作符

    左边丢弃，右边补0

##### 正数：

    int main()
    {
        int a = 5;//4byte = 32bit = 
        正整数在内存中的存储：    
            源码：00000000000000000000000000000101
    	    反码：00000000000000000000000000000101
    	    补码：00000000000000000000000000000101 (内存)(移动)
    	    
    	int b = a << 1;
    	printf("%d\n", b);//10
    	所有0向左移动1位，右面补0
            源码：00000000000000000000000000001010
            反码：00000000000000000000000000001010
            补码：00000000000000000000000000001010 (内存)(移动)
    
        return 0;
    }

##### 负数：

    int main()
    {
        int c = -1;
        负整数在内存中的存储
            源码：10000000000000000000000000000001 (打印)
    	    反码：11111111111111111111111111111110
    	    补码：11111111111111111111111111111111 (内存)(移动)
    	    
    	int d = c << 1;
    	printf("%d\n", d);//-2 - 打印的是原码的值
            补码：11111111111111111111111111111110 (内存)(移动)
            反码：11111111111111111111111111111101
            源码：10000000000000000000000000000010 (打印)
       
        return 0;
    }

#### >> ：右移操作符

##### 1.逻辑右移 - 右边丢弃，左边补0
##### 2.算数右移 - 右边丢弃，左边补原符号位

##### 正数：

    int main()
    {
        int a = 5;
        int b = a >> 1; //2
        逻辑右移、算数右移一样
        移动后：
            源码：00000000000000000000000000000010
            反码：00000000000000000000000000000010
            补码：00000000000000000000000000000010 (内存)(移动)
        return 0;
    }

##### 负数： vs 采用的算数右移

    int main()
    {
        int c = -1;
        负整数在内存中的存储
            源码：10000000000000000000000000000001 (打印)
    	    反码：11111111111111111111111111111110
    	    补码：11111111111111111111111111111111 (内存)(移动)
    	    
    	int d = c << 1;
    	printf("%d\n", d);//-1 - 当前编译器采用算数右移
    	算数右移后：不变(右边丢弃，左边补原符号位)
            补码：11111111111111111111111111111111 (内存)(移动)
        逻辑右移后：正数(右边丢弃，左边补0)
            补码：01111111111111111111111111111111 (内存)(移动)
            反码：01111111111111111111111111111101
            源码：10000000000000000000000000000010 (打印)
       
        return 0;
    }

#  位操作符：与或非

##  &   |   ^ 

- &


    按位与&：  两个数的补码按位与  同1为1  有0为0
    正整数 原反补相同
    负数    写源码  源码
            取反    反码
            +1      补码
    
    按位或|：    两个数的补码按位或 有1为1
    得到的补码符号位是1 为负数
    打印的时候打印的是源码
    符号位不变  补码-1 取反 得到源码
    
    按位异或^：  两个数的补码按位异或 同为0  异为1
    方法同上

- 实现两个数的交换


    a = 3
    b = 5
    
    1.  tmp = 0
        tmp = a
        a = b
        b = tmp
        --------用的最多
    
    2.  a = a+b
        b = a-b
        a = a-b
        --------不好：会溢出
        
    3.  a = a ^ b
        b = a ^ b
        a = a ^ b
        --------用的少 代码可读性差  只适用于整形
        
        原理：
            a ^ a = 0
            a ^ 0 = a
            
            a ^ a ^ b = b
            a ^ a ^ a = a
            
            a = a ^ b
            b = (a ^ b) ^ b => a
            a = (a ^ b) ^ a => b
            
        结论：
            异或支持交换律

# 下标引用

### 整形提升


    负数符号位补1
    正数符号位补0

    int main()
    {
    	char a = 3;//a为char类型  1byte = 8bit
    	//00000000000000000000000000000011
    	//00000011 - a  截断
    	//00000011 - 第一个0 为符号位
    	char b = 127;
    	//00000000000000000000000001111111
    	//01111111 - a
    	
    	//a 和b 都是char类型，1byte  计算式整形提升
    	//00000000000000000000000000000011 - a
    	//00000000000000000000000001111111 - b
    	//00000000000000000000000010000010 - 计算
    	char c = a + b; 
    	//10000010 - c
    	//11111111111111111111111110000010 - 补码
    	//11111111111111111111111110000001 - 补码-1 =》反码 
    	//10000000000000000000000001111110 - 取反   =》源码
    	//-126
    	printf("%d", c);
    	return 0;
    }

## 操作符的属性

    1.操作符的优先级
    2.操作符的结合性
    3.是否控制求值顺序
    
    表达式1
    a*b + c*d + e*f
    1   4  2  5  3
    1   3  2  5  4
    不同的顺序会 可能 计算出不同的结果
    
    表达式2 
    c + --c
    先++ --   再+ -
    不能判断c 是在 --c之前就存在  还是--c计算完后再存在
    
    写成  
        int c = 3;
        int e = --c;
        int d = c + e;


    int main()
    {

    }
