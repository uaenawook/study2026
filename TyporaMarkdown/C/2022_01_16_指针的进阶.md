## 1.字符指针

```c
    #include <stdio.h>
    int main()
    {
    	char ch = 'w';
    	char* pc = &ch; //pc是指向一个字符的变量
    
    	const char* p = "hello bit";//"hello bit"常量字符串
    	//把常量字符串"hello bit"的第一个字符h的地址赋值给p
    
    	//*p = 'w';//常量字符串不能修改
    	printf("%c\n", *p);
    	printf("%s\n", *p);
    	return 0;

    %c打印*p 得到h  因为p存的h的地址
    %s打印p得到hello bit 因为打印字符串给首字母地址就可以了 
        如果解引用就打印不出来了
```

## 3.数组指针
> 指向数组的指针


    数组指针是指针？还是数组？
    答案是：指针。

```
    int *p1[10];//指针数组
    int (*p2)[10];//数组指针
```
```c
    int arr[10];
    arr - 数组首元素的地址
    &arr - 数组的地址
```
**书写方式**


    int main()
    {
    	int* arr[10] = { 0 };
    	int* (*p)[10] = &arr;//数组指针
    	int** p1 = arr;//二级指针
    	int** p2 = &arr[0];//二级指针
    	printf("%p\n", p);//0113FC58
    	printf("%p\n", p1);//0113FC58
    	printf("%p\n", p2);//0113FC58
        
        地址的大小 取决于 指针的类型 
        int* 是arr数组每个元素的 类型
        
        int* (*)[10] 是指向arr数组的指针的类型
            (*)证明p是指针
            [10]说明 p指向的数组是10个元素
            int* 说明 p 指向数组的类型是int*
                
    	printf("%p\n", p+1);//0113FC80  - 40  
    	printf("%p\n", p1+1);//0113FC5C - 4  
    	printf("%p\n", p2+1);//0113FC5C - 4  
    	return 0;
    }
