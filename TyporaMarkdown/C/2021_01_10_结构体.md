## 1.结构类型的声明
#### 1.1结构的基础知识


    结构体是一些值得集合 这些值称为成员变量
    结构的每个成员可以是不同类型的变量
#### 1.2结构的声明
- **struct**


    struct tag
    {
        member-list；
    }variable-list；  这里创建的变量是全局变量
    
    如果以上放在main函数中   则variable-list 为局部变量

#### 1.3结构体成员的类型

    变量、数组、指针、其他结构体

#### 1.4结构体变量的定义和初始化

- 描述复杂对象  - 定义
  
  
    1.书
    struct Book
    {
    	char name[20];
    	char author[15];
    	float price;
    };
    
    2.坐标
    struct Point

    	int x;

    };
        
    int main()
    {
    	struct Book b;//局部变量 - 栈区
    
    	return 0;

  typedef 定义结构体


    typedef struct Book

## 2.结构体初始化


    struct Book b = {"C语言","20211222",30}；
    Book b = 0;

- 嵌套初始化


    struct S
    {
        int a;
        char b;
    };
    struct T
    {
        struct S s;
        int c;
        char b;
    }
    struct T t = {{20,'c'},50,"abab"}

- 访问结构体变量


    struct S
    {
    	char name[20];
    	int age;
    };
    struct T
    {
    	struct S s;
    	char name[20];
    	int age;
    };
    int main()
    {
    	struct T t = { {"张三",20},"李四",22 };
    	printf("%s %d %s %d\n", t.s.name,t.s.age,t.name,t.age);//张三 20 李四 22
    	struct T* pt = &t;
    	printf("%s %d %s %d\n", pt->s.name, pt->s.age, pt->name, pt->age);//张三 20 李四 22
    	return 0;
    }
## 3.结构体传参


    struct S
    {
    	int arr[100];
    	char name[20];
    	int age;
    };
    void print2(struct S* ps) 传址调用
    {
    	printf("%d %d %d %s %d\n", ps->arr[0], ps->arr[1], ps->arr[2], ps->name, ps->age);//1 2 3 zhangsan 20
    }
    void print1(struct S ss) 传值调用 需要压栈  浪费空间
    {
    	printf("%d %d %d %s %d\n", ss.arr[0], ss.arr[1], ss.arr[2], ss.name, ss.age);//1 2 3 zhangsan 20
    }
    int main()
    {
    	struct S s = { {1,2,3},"zhangsan",20 };
    	print1(s);
    	print2(&s);
    
    	return 0;
    }
