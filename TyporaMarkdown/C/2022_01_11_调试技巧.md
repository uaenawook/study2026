    ### 调试的重要性
    
        以下代码会出现死循环 为什么呢？
        int main()
        {
        	int i = 0;
        	int arr[10] = { 1,2,3,4,5,6,7,8,9,0 };
        
        	for (i = 0; i <= 12; i++)
        	{
        		arr[i] = 0;
        		printf("hehe\n");
        	}
        	return 0;
        }
        
        可以看出以上代码出现了越界访问 
        通过调试->监视可以发现 arr[12]赋值为0时
        i也被赋值为0；
        通过查看地址可以发现 i的地址和 arr[12]相同
        也就是说arr[12]把i赋值为了0；
        
        解释起来有三点
        1.局部变量放在栈区里
        2.栈区的使用习惯（高地址->低地址）
            低
                arr[0]
                arr[1]
                ...
                arr[9]
                
                
                i
            高
            因为程序先创建的i  所以i在高地址
            后创建的arr 所以arr在低地址
            当arr由低到高访问时 就有可能访问到i 并把i重置
            造成死循环
            
            我们知道：数组元素的地址是由低到高变化的
            
            解决这个问题有两种方法
                1.尽量避免越界访问
                2.把i放在arr数组后创建
                    这样arr在高地址  i在低地址
                    arr向高地址访问就不会 访问到i
            
            当for循环改成for(i = 0;i<=10;i++)就会报错
            为什么访问到 12的时候没报错？
                因为程序 访问到最后一个元素的时候 运行完成才会判断错误
                当访问到10的时候程序结束了  所以直接报错
                当访问到12的时候程序还在死循环  程序还在进行 没时间报错
            
        3.随着下标的越界可能导致死循环

         const 放* 右面 修饰p 不能改
         *左右都可以放const

        {
        	const int num = 10;
        ------------------------------
        	const  修饰指针
        	const  放在* 的左边
        	------------------------------
        	修饰的是指针指向的内容（*p），指针指向的内容，不能通过指针来改变了
        	但是指针变量本身是可以改变的
        	 
        	int n = 100;
        	const int* p = &num;  // int const * p = &num
        	*p = 20;
        	p = &n;
            ------------------------------	
        	const  放在* 的右边
        	修饰的是指针变量本身（p），指针变量本身不能修改，
        	但是指针指向的内容（*p）可以修改
        	int n = 100;
        	int* const p = &num;
        	*p = 20;//ok
        	p = &n;//err
        ------------------------------
        	return 0;
        }

        size_t  unsigned int   无符号整形
        无符号-无符号  = 无符号   不会出现负数
        unsigned int my_strlen(const char* arr)
        {
        	assert(arr);
        	int num = 0;
        	while (*arr++)
        	{
        		num++;
        	}
        	return num;
        }
        int main()
        {
        	char arr[] = "abcde";

    - 求一个十进制数字的二进制有几个1
    -
        
        n = n&(n-1)
        
        例如：
        15 的二进制1111;
        
        1111 - n
        1110 - n-1;
        &完 = 1110
        
        1110 - n
        1101 - n-1;
        &完 = 1100
        
        1100 - n
        1011 - n - 1;
        &完 = 1000;
        
        1000 - n
        0111 - n-1;
        &完 = 0000 
        
        以上代码共执行了4次  所以有4个1
        如果是负数 则传参时用 unsigned 无符号整形
        
    - 用以上方法可以算出 n是不是2的次方数
    -    
    
        if(n&(n-1) == 0)
        {
            n就是2的次方
        }
        因为
        2       0010
        4       0100
